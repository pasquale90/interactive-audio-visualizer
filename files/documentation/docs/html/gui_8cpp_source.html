214void GUI::initializeTexts(){
215
216 // Get audio devices and sample rates supported
217 std::vector<AudioHardware::Info> audio_hw_info;
218 get_audio_hardware_info(audio_hw_info);
219 for (const auto&[info,sr,nChannels]:audio_hw_info){
220 // printf("%s : %d, %d\n",name.c_str(), sr.first, sr.second);
221 QString audio_device = QString::fromStdString(info.first);
222 audioExplanations.push_back(info.second);
223 audioDevices.append(audio_device);
224 numChannels.append(QString::number(nChannels));
225 for (auto srate: AudioHardware::supportedRates){
226 if (srate >= sr.first && srate <= sr.second){
227 sampleRates[audio_device].append(QString::number(srate));
228 }
229 }
230 }
231
232 // Get available cameras and resolutions supported for each one of them
233 auto cameras = getAvailableCameras();
234 for (const auto& camera : cameras) {
235
236 QString camera_device = QString::fromStdString(camera.devicePath);
237
238 if (!camera.resolutions.empty()){
239 cameraDevices.append(camera_device);
240 for (const auto& res : camera.resolutions) {
241 cameraResolutions[camera_device].append(QString::number(res.first) + "x" + QString::number(res.second));
242 }
243 }
244 }
245
246 // Get screen resolutions for the main screen
247 auto screen_resolutions = get_screen_resolution();
248 for(const auto&resolution : screen_resolutions){
249 displayResolutions.append(QString::number(resolution.first) + "x" + QString::number(resolution.second));
250 }
251

361bool GUI::checkResolutionCompatibility() {
362 // Resolutions are currently stored as "WidthxHeight"
363 QString cameraRes = resolutionComboBox->currentText();
364 QString displayRes = displayResolutionComboBox->currentText();
365
366 int cameraWidth = cameraRes.split("x")[0].toInt();
367 int cameraHeight = cameraRes.split("x")[1].toInt();
368 int displayWidth = displayRes.split("x")[0].toInt();
369 int displayHeight = displayRes.split("x")[1].toInt();
370
371 return (cameraWidth <= displayWidth && cameraHeight <= displayHeight);
372}
