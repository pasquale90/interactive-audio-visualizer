262void Visualizer::draWaveform(){
263
264 int W=cfg.dispconf.dispResW.load();
265 int H=cfg.dispconf.dispResH.load();
266 int cameraW = cfg.camconf.camResW.load();
267 int cameraH = cfg.camconf.camResH.load();
268
269 int x_centre = W/2;
270 int y_centre = H/2;
271 int r = (cameraW>cameraH) ? cameraH/2 : cameraW/2;
272
273 float minMax[2];
274 size_t numSamples;
275 waveform.getMinMax(minMax);
276 float min = minMax[0],max = minMax[1];
277 numSamples = waveform.availableForReading();
278
279
280 // depict waveform
281 size_t end;
282 double curRadians=0.0;
283 double radianStep=2*M_PI / (double)numPointsPerimeter;
284
285 if (numSamples< static_cast<size_t>(numPointsPerimeter)){
286 end=numSamples;
287 }else end = static_cast<size_t>(numPointsPerimeter);
288
289 int thickness=1;
290 int x1,x2,y1,y2;
291 float percent;
292 float new_radius;
293 float waveVal;
294 for (size_t i=0; i<end ; i++){
295
296 // pixels are calculated given the following equations:
297 // x = cx + r * cos(a)
298 // y = cy + r * sin(a)
299 x1 = static_cast<int>((float)r * std::cos(curRadians) + (float)x_centre);
300 y1 = static_cast<int>((float)r * std::sin(curRadians) + (float)y_centre);
301
302 waveform.read(waveVal);
303 // normalize in range [-1 1]
304 percent = 2* ( (waveVal-min) / (max-min) ) -1;
305
306 // trasport x and y
307 new_radius = static_cast<float>(r + ((float)H / 40. * percent));
308
309 x2 = static_cast<int>(new_radius * std::cos(curRadians) + (float)x_centre);
310 y2 = static_cast<int>(new_radius * std::sin(curRadians) + (float)y_centre);
311
312 cv::Point p1(x1, y1), p2(x2, y2);
313 cv::line(visualFrame, p1, p2, cv::Scalar(255, 0, 0), thickness, cv::LINE_8);
314
315 curRadians+=radianStep;
316 }

376void Visualizer::drawSpectrogram(){
377
378 const double FREQ_MIN = (double)cfg.iavconf.minFrequency;
379 const double FREQ_MAX = (double)cfg.iavconf.maxFrequency;
380 int SR = cfg.audconf.sampleRate.load();
381 int W = cfg.dispconf.dispResW.load();
382 int H = cfg.dispconf.dispResH.load();
383 int numAudioSamples = spectrogram.get_numAudioSamples();
384
385 float min_magnitude,max_magnitude;
386 spectrogram.readBatch(specMagnitude,min_magnitude,max_magnitude);
387
388 for (size_t i = 0; i < specMagnitude.size() ; ++i) {
389
390 float magnitude = specMagnitude[i];
391
392 // Minmax magnitude's normalization
393 int normalized_magnitude = static_cast<int>(
394 std::min(
395 (float)((magnitude - min_magnitude) / (max_magnitude - min_magnitude) * static_cast<float>(H) * 0.5),
396 static_cast<float>(H)/2.f
397 )
398 );
399
400 // Calculate the frequency interval for each bin
401 double freq_bin = (SR / 2.0) * (static_cast<double>(i) / static_cast<double>(numAudioSamples / 2.0)); // Frequency of the current bin
402
403 // filtering the frequencies outside the scope of the iav application
404 if (freq_bin >= FREQ_MIN && freq_bin <= FREQ_MAX) {
405
406 int x = static_cast<int>((freq_bin - FREQ_MIN) / (FREQ_MAX - FREQ_MIN) * W);
407
408 // Create a the drawing line
409 int line_length = normalized_magnitude;
410 int top = H / 2 - line_length;
411 int bottom = H / 2 + line_length;
412
413 for (int y = top; y <= bottom; ++y) {
414 // Ensure not out-of-bounds
415 if (y >= 0 && y < H && x >= 0 && x < W) {
416 visualFrame.at<cv::Vec3b>(y,x)[0] = 255 - B;
417 visualFrame.at<cv::Vec3b>(y,x)[1] = 255 - G;
418 visualFrame.at<cv::Vec3b>(y,x)[2] = 255 - R;

420 }
421 }

423 }
